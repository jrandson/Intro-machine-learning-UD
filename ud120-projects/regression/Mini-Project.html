import numpy as np

def remove_outlier(data):
    data = np.sort(data)
    #print '------------------------'
    
    h = len(data)/2
    data1 = data[:h]
    data2 = data[h:]
    
    q1 = np.median(data1)
    q3 = np.median(data2)
    IQR = q3 - q1
    
    #print "Q1:", q1
    #print "Q3:", q3
    #print "IQR:", IQR
    
    limit_min = q1 - 1.5 * IQR
    limit_max = q3 + 1.5 * IQR
    
    data_ = data[data > limit_min]
    data_ = data_[data_ < limit_max]
    
    #print '------------------------'
    
    return data_
